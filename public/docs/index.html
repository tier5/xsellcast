    Brand

    All

    var settings = {
    "async": true,
    "crossDomain": true,

    Delete Offer

    Delete an offer related to customer. The @parameter $_method is required and value must set to DELETE.

    Example request:

    curl "http://xsellcast.local/api/v1/customer/offer" \
-H "Accept: application/json" \
    -d "access_token"="neque" \
    -d "customer_id"="13" \
    -d "offer_id"="13" \
    -d "_method"="DELETE" \

    var settings = {
    "async": true,
    "crossDomain": true,
    "url": "http://xsellcast.local/api/v1/customer/offer",
    "method": "DELETE",

        "access_token": "neque",
        "customer_id": 13,
        "offer_id": 13,
        "_method": "DELETE"
},
    "headers": {
        "accept": "application/json"
